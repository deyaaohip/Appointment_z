    57→// PAGE 1: OVERVIEW (Audited — all KPIs derived from real data)
    58→// ════════════════════════════════════════════════════════════════
    59→function OverviewPage() {
    60→  const { t, isRTL, lang } = useSA()
    61→  const { sym, fmt } = useCurrency()
    62→  const [tenants] = useState<Tenant[]>(INIT_TENANTS)
    63→  const [logs] = useState<SysLog[]>(INIT_LOGS)
    64→  const [invoices] = useState<Invoice[]>(INVOICES)
    65→
    66→  // Computed from real data — no hardcoded values
    67→  const totalRevenue = tenants.reduce((s, tn) => s + tn.revenue, 0)
    68→  const totalBookings = tenants.reduce((s, tn) => s + tn.bookings, 0)
    69→  const activeCount = tenants.filter(tn => tn.status === 'active').length
    70→  const expiredCount = tenants.filter(tn => tn.subscriptionStatus === 'expired').length
    71→  const pendingSubs = tenants.filter(tn => tn.subscriptionStatus === 'trial' || tn.status === 'trial').length
    72→  const paidInvoices = invoices.filter(i => i.status === 'paid')
    73→  const monthlyRevenue = paidInvoices.reduce((s, i) => s + i.amount, 0)
    74→  const activeTenants = tenants.filter(tn => tn.status === 'active').slice(0, 6)
    75→
    76→  return (
    77→    <div className="space-y-6">
    78→      {/* Primary KPIs — all computed from tenant/invoice data */}
    79→      <div className="grid grid-cols-2 lg:grid-cols-4 gap-3 sm:gap-4">
    80→        <KpiCard icon={Building2} bg="bg-violet-600" label={t.totalTenants} value={tenants.length} sub={`${activeCount} ${t.active.toLowerCase()}`} trend={12} delay={0} />
    81→        <KpiCard icon={Users} bg="bg-sky-600" label={t.totalUsers} value={INIT_USERS.length} sub={`${tenants.reduce((s, tn) => s + tn.users, 0)} ${t.users.toLowerCase()}`} trend={8} delay={0.04} />
    82→        <KpiCard icon={DollarSign} bg="bg-amber-500" label={t.revenue} value={`${(totalRevenue / 1000).toFixed(1)}K ${sym}`} sub={t.thisMonth} trend={18} delay={0.08} />
    83→        <KpiCard icon={CalendarDays} bg="bg-emerald-600" label={t.totalBookings} value={totalBookings.toLocaleString()} sub={t.last30Days} trend={5} delay={0.12} />
    84→      </div>
    85→
    86→      {/* Secondary KPIs — subscription health */}
    87→      <div className="grid grid-cols-2 lg:grid-cols-4 gap-3 sm:gap-4">
    88→        <StatCard icon={CheckCircle2} label={t.monthlyRevenue} value={fmt(monthlyRevenue)} color="text-emerald-600 bg-emerald-50 dark:bg-emerald-950/30" />
    89→        <StatCard icon={Clock} label={t.pendingRequests} value={String(pendingSubs)} color="text-amber-600 bg-amber-50 dark:bg-amber-950/30" />
    90→        <StatCard icon={AlertTriangle} label={t.expiredSubs} value={String(expiredCount)} color="text-red-600 bg-red-50 dark:bg-red-950/30" />
    91→        <StatCard icon={CreditCard} label={t.recentPayments} value={String(paidInvoices.length)} color="text-sky-600 bg-sky-50 dark:bg-sky-950/30" />
    92→      </div>
    93→
    94→      <div className="grid grid-cols-1 xl:grid-cols-3 gap-4 sm:gap-6">
    95→        <motion.div variants={fade} initial="hidden" animate="visible" transition={{ delay: 0.15 }} className="xl:col-span-2">
    96→          <Card className="border-0 shadow-sm">
    97→            <CardHeader className="pb-2"><CardTitle className="text-sm sm:text-base font-semibold">{t.revenueByPlan}</CardTitle></CardHeader>
    98→            <CardContent className="space-y-4">
    99→              {PLANS.filter(p => p.price > 0).map(p => {
   100→                const rev = tenants.filter(tn => tn.plan === p.name).reduce((s, tn) => s + tn.revenue, 0)
   101→                const pct = totalRevenue > 0 ? Math.round(rev / totalRevenue * 100) : 0
   102→                return (
   103→                  <div key={p.id} className="space-y-2">
   104→                    <div className="flex items-center justify-between text-sm">
   105→                      <span className="font-medium">{p.name}</span>
   106→                      <span className="text-muted-foreground tabular-nums text-xs sm:text-sm">{rev.toLocaleString()} {sym}</span>
   107→                    </div>
   108→                    <div className="h-2.5 rounded-full bg-muted overflow-hidden">
   109→                      <div className={`h-full rounded-full ${p.color} transition-all duration-700`} style={{ width: `${pct}%` }} />
   110→                    </div>
   111→                  </div>
   112→                )
   113→              })}
   114→            </CardContent>
   115→          </Card>
   116→        </motion.div>
   117→
   118→        <motion.div variants={fade} initial="hidden" animate="visible" transition={{ delay: 0.2 }}>
   119→          <Card className="border-0 shadow-sm h-full">
   120→            <CardHeader className="pb-2"><CardTitle className="text-sm sm:text-base font-semibold">{t.recentActivity}</CardTitle></CardHeader>
   121→            <CardContent>
   122→              <div className="space-y-1">{logs.slice(0, 6).map(l => (
   123→                <div key={l.id} className="flex gap-3 rounded-lg px-2.5 py-2 hover:bg-muted/50 transition-colors">
   124→                  <LogDot level={l.level} />
   125→                  <div className="min-w-0 flex-1">
   126→                    <p className="text-xs sm:text-sm leading-relaxed line-clamp-2">{logMsg(l, lang)}</p>
   127→                    <p className="text-[10px] sm:text-[11px] text-muted-foreground mt-0.5">{logSrc(l, lang)}</p>
   128→                  </div>
   129→                </div>
   130→              ))}</div>
   131→            </CardContent>
   132→          </Card>
   133→        </motion.div>
   134→      </div>
   135→
   136→      <motion.div variants={fade} initial="hidden" animate="visible" transition={{ delay: 0.25 }}>
   137→        <Card className="border-0 shadow-sm">
   138→          <CardHeader className="pb-2"><CardTitle className="text-sm sm:text-base font-semibold">{t.activeTenants}</CardTitle></CardHeader>
   139→          <CardContent>
   140→            <div className="grid grid-cols-1 sm:grid-cols-2 lg:grid-cols-3 gap-2 sm:gap-3">
   141→              {activeTenants.map(tn => (
   142→                <button key={tn.id} className="flex items-center gap-3 rounded-xl border p-3 text-start hover:bg-accent/50 transition-colors group" onClick={() => toast.info(t.details.replace('{name}', bField(tn, 'name', lang)))}>
   143→                  <div className="flex h-10 w-10 shrink-0 items-center justify-center rounded-lg bg-violet-100 dark:bg-violet-900/30 text-violet-600 font-bold text-sm">{bField(tn, 'name', lang).charAt(0)}</div>
   144→                  <div className="min-w-0 flex-1">
   145→                    <p className="text-sm font-semibold truncate">{bField(tn, 'name', lang)}</p>
   146→                    <p className="text-xs text-muted-foreground">{tn.plan} · {tn.bookings.toLocaleString()} {t.bookings}</p>
   147→                  </div>
   148→                  <ChevronLeft className="h-4 w-4 text-muted-foreground shrink-0 opacity-0 group-hover:opacity-100 transition-opacity" style={{ transform: isRTL ? 'scaleX(-1)' : undefined }} />
   149→                </button>
   150→              ))}
   151→            </div>
   152→          </CardContent>
   153→        </Card>
   154→      </motion.div>
   155→    </div>

   160→// PAGE 2: TENANTS (full CRUD + sort + pagination)
   161→// ════════════════════════════════════════════════════════════════
   162→function TenantsPage() {
   163→  const { t, isRTL, lang } = useSA()
   164→  const { sym } = useCurrency()
   165→  const [tenants, setTenants] = useState<Tenant[]>(INIT_TENANTS)
   166→  const [search, setSearch] = useState('')
   167→  const [filter, setFilter] = useState('all')
   168→  const [page, setPage] = useState(1)
   169→  const [sort, setSort] = useState<SortState>({ key: 'name', dir: 'asc' })
   170→  const [dlg, setDlg] = useState<{ type: string; tenant?: Tenant } | null>(null)
   171→  const [form, setForm] = useState({ name: '', nameEn: '', email: '', country: '', plan: 'Starter' })
   172→  const [extendDlg, setExtendDlg] = useState<Tenant | null>(null)
   173→  const [extendMonths, setExtendMonths] = useState(12)
   174→  const [extendPlan, setExtendPlan] = useState('')
   175→
   176→  const filtered = useMemo(() => {
   177→    let list = [...tenants]
   178→    if (filter !== 'all') list = list.filter(tn => tn.status === filter)
   179→    if (search) { const q = search.toLowerCase(); list = list.filter(tn => tn.name.includes(q) || tn.nameEn.toLowerCase().includes(q) || tn.email.toLowerCase().includes(q)) }
   180→    return genericSort(list, sort.key, sort.dir, lang)
   181→  }, [tenants, search, filter, sort, lang])
   182→
   183→  const { items, totalPages } = paginate(filtered, page, PER_PAGE)
   184→  const handleSort = (key: string) => setSort(p => p.key === key ? { key, dir: p.dir === 'asc' ? 'desc' : 'asc' } : { key, dir: 'asc' })
   185→
   186→  const openAdd = () => { setForm({ name: '', nameEn: '', email: '', country: '', plan: 'Starter' }); setDlg({ type: 'add' }) }
   187→  const openEdit = (tn: Tenant) => { setForm({ name: tn.name, nameEn: tn.nameEn, email: tn.email, country: tn.country, plan: tn.plan }); setDlg({ type: 'edit', tenant: tn }) }
   188→
   189→  const handleSave = useCallback(() => {
   190→    if (!form.name.trim()) { toast.error(t.enterTenantName); return }
   191→    if (form.email && !form.email.includes('@')) { toast.error(t.enterValidEmail); return }
   192→    if (dlg?.type === 'add') {
   193→      const n: Tenant = { id: Date.now().toString(), ...form, bookings: 0, revenue: 0, users: 1, status: 'trial', createdAt: new Date().toISOString().split('T')[0], subscriptionStatus: 'trial', subscriptionEndDate: '' }
   194→      setTenants(p => [n, ...p]); toast.success(t.added.replace('{name}', form.name))
   195→    } else if (dlg?.tenant) {
   196→      setTenants(p => p.map(tn => tn.id === dlg.tenant!.id ? { ...tn, ...form } : tn)); toast.success(t.updated)
   197→    }
   198→    setDlg(null); setPage(1)
   199→  }, [form, dlg, t])
   200→
   201→  const handleDelete = (tn: Tenant) => { setTenants(p => p.filter(x => x.id !== tn.id)); toast.success(t.deleted); setPage(1) }
   202→  const handleToggleStatus = (tn: Tenant) => {
   203→    const newStatus = tn.status === 'suspended' ? 'active' : 'suspended'
   204→    setTenants(p => p.map(x => x.id === tn.id ? { ...x, status: newStatus } : x))
   205→    toast.success(newStatus === 'suspended' ? t.tenantSuspended : t.tenantActivated)
   206→    setPage(1)
   207→  }
   208→
   209→  const openExtend = (tn: Tenant) => { setExtendDlg(tn); setExtendPlan(tn.plan); setExtendMonths(12) }
   210→  const handleExtend = () => {
   211→    if (!extendDlg) return
   212→    const startDate = extendDlg.subscriptionEndDate && extendDlg.subscriptionEndDate > new Date().toISOString().split('T')[0]
   213→      ? new Date(extendDlg.subscriptionEndDate) : new Date()
   214→    const newEnd = new Date(startDate)
   215→    newEnd.setMonth(newEnd.getMonth() + extendMonths)
   216→    setTenants(p => p.map(x => x.id === extendDlg.id ? {
   217→      ...x, plan: extendPlan || x.plan,
   218→      subscriptionStatus: 'active', status: 'active',
   219→      subscriptionEndDate: newEnd.toISOString().split('T')[0]
   220→    } : x))
   221→    toast.success(t.subscriptionExtended)
   222→    setExtendDlg(null)
   223→  }
   224→
   225→  const tnName = (tn: Tenant) => bField(tn, 'name', lang)
   226→
   227→  return (
   228→    <div className="space-y-5">
   229→      <PageTitle title={t.tenantMgmt} action={<PrimaryBtn icon={Plus} label={t.addTenant} onClick={openAdd} />} />
   230→      <div className="flex flex-col sm:flex-row gap-3">
   231→        <SearchInput value={search} onChange={v => { setSearch(v); setPage(1) }} placeholder={t.tenantSearch} />
   232→        <Select value={filter} onValueChange={v => { setFilter(v); setPage(1) }}>
   233→          <SelectTrigger className="w-full sm:w-44 h-10"><SelectValue /></SelectTrigger>
   234→          <SelectContent>
   235→            <SelectItem value="all">{t.allStatuses}</SelectItem>
   236→            <SelectItem value="active">{t.active}</SelectItem>
   237→            <SelectItem value="trial">{t.trial}</SelectItem>
   238→            <SelectItem value="suspended">{t.suspended}</SelectItem>
   239→          </SelectContent>
   240→        </Select>
   241→      </div>
   242→
   243→      <Card className="border-0 shadow-sm overflow-hidden hidden md:block">
   244→        <div className="overflow-x-auto">
   245→          <Table>
   246→            <TableHeader><TableRow className="bg-muted/50 hover:bg-muted/50">
   247→              <SortableTH label={t.tenant} sortKey="name" currentSort={sort} onSort={handleSort} className="ps-5" />
   248→              <SortableTH label={t.plan} sortKey="plan" currentSort={sort} onSort={handleSort} />
   249→              <SortableTH label={t.bookings} sortKey="bookings" currentSort={sort} onSort={handleSort} align="end" />
   250→              <SortableTH label={t.tenantRevenue} sortKey="revenue" currentSort={sort} onSort={handleSort} align="end" />
   251→              <SortableTH label={t.status} sortKey="status" currentSort={sort} onSort={handleSort} />
   252→              <TableHead className="px-3 font-semibold text-xs uppercase tracking-wider">{t.subscriptionStatus}</TableHead>
   253→              <TableHead className="pe-5 font-semibold text-xs uppercase tracking-wider text-end">{t.actions}</TableHead>
   254→            </TableRow></TableHeader>
   255→            <TableBody>
   256→              {items.length === 0 ? <EmptyRow colSpan={7} /> : items.map(tn => (
   257→                <TableRow key={tn.id} className="group hover:bg-muted/20 transition-colors">
   258→                  <TableCell className="ps-5">
   259→                    <div className="flex items-center gap-3">
   260→                      <div className="flex h-9 w-9 shrink-0 items-center justify-center rounded-lg bg-violet-100 dark:bg-violet-900/30 text-violet-600 font-bold text-xs">{tnName(tn).charAt(0)}</div>
   261→                      <div className="min-w-0"><p className="font-medium text-sm truncate max-w-[200px]">{tnName(tn)}</p><p className="text-xs text-muted-foreground truncate max-w-[200px]" dir="ltr">{tn.email}</p></div>
   262→                    </div>
   263→                  </TableCell>
   264→                  <TableCell className="px-3"><Badge variant="secondary" className="font-medium text-xs">{tn.plan}</Badge></TableCell>
   265→                  <TableCell className="px-3 text-end tabular-nums text-sm">{tn.bookings.toLocaleString()}</TableCell>
   266→                  <TableCell className="px-3 text-end font-semibold tabular-nums text-sm">{tn.revenue.toLocaleString()} <span className="text-xs font-normal text-muted-foreground">{sym}</span></TableCell>
   267→                  <TableCell className="px-3"><StatusBadge status={tn.status} locale={lang} /></TableCell>
   268→                  <TableCell className="px-3">
   269→                    <div className="space-y-0.5">
   270→                      <StatusBadge status={tn.subscriptionStatus} locale={lang} />
   271→                      {tn.subscriptionEndDate && <p className="text-[10px] text-muted-foreground">{t.subscriptionEnd}: {tn.subscriptionEndDate}</p>}
   272→                    </div>
   273→                  </TableCell>
   274→                  <TableCell className="pe-5">
   275→                    <div className="flex items-center justify-end gap-1 opacity-0 group-hover:opacity-100 transition-opacity">
   276→                      <ActionBtn icon={Eye} label={t.view} onClick={() => toast.info(t.details.replace('{name}', tnName(tn)))} />
   277→                      <ActionBtn icon={Edit} label={t.edit} onClick={() => openEdit(tn)} />
   278→                      <ActionBtn icon={CalendarDays} label={t.extendSubscription} onClick={() => openExtend(tn)} />
   279→                      <ActionBtn icon={tn.status === 'suspended' ? Power : PowerOff} label={tn.status === 'suspended' ? t.active : t.suspended} onClick={() => handleToggleStatus(tn)} />
   280→                      <ActionBtn icon={Trash2} onClick={() => setDlg({ type: 'delete', tenant: tn })} danger />
   281→                    </div>
   282→                  </TableCell>
   283→                </TableRow>
   284→              ))}
   285→            </TableBody>
   286→          </Table>
   287→        </div>
   288→        <TableFooter current={items.length} total={filtered.length} entity={t.tenant.toLowerCase()} />
   289→        <Pagination page={page} totalPages={totalPages} onPageChange={setPage} />
   290→      </Card>
   291→
   292→      <div className="md:hidden space-y-3">
   293→        {items.length === 0 ? <Card className="border-0 shadow-sm"><CardContent className="py-16 text-center text-muted-foreground text-sm">{t.noResults}</CardContent></Card> : items.map(tn => (
   294→          <Card key={tn.id} className="border-0 shadow-sm"><CardContent className="p-4">
   295→            <div className="flex items-start justify-between gap-3 mb-3">
   296→              <div className="flex items-center gap-3 min-w-0">
   297→                <div className="flex h-10 w-10 shrink-0 items-center justify-center rounded-lg bg-violet-100 dark:bg-violet-900/30 text-violet-600 font-bold text-sm">{tnName(tn).charAt(0)}</div>
   298→                <div className="min-w-0"><p className="font-semibold text-sm truncate">{tnName(tn)}</p><p className="text-xs text-muted-foreground truncate" dir="ltr">{tn.email}</p></div>
   299→              </div>
   300→              <StatusBadge status={tn.status} locale={lang} />
   301→            </div>
   302→            <div className="grid grid-cols-3 gap-3 mb-3 text-center">
   303→              <div className="rounded-lg bg-muted/50 p-2"><p className="text-[10px] text-muted-foreground">{t.plan}</p><p className="text-xs font-semibold mt-0.5">{tn.plan}</p></div>
   304→              <div className="rounded-lg bg-muted/50 p-2"><p className="text-[10px] text-muted-foreground">{t.bookings}</p><p className="text-xs font-semibold mt-0.5 tabular-nums">{tn.bookings.toLocaleString()}</p></div>
   305→              <div className="rounded-lg bg-muted/50 p-2"><p className="text-[10px] text-muted-foreground">{t.tenantRevenue}</p><p className="text-xs font-semibold mt-0.5 tabular-nums">{tn.revenue.toLocaleString()} {sym}</p></div>
   306→            </div>
   307→            <div className="flex items-center gap-2 mb-3">
   308→              <StatusBadge status={tn.subscriptionStatus} locale={lang} />
   309→              {tn.subscriptionEndDate && <span className="text-[10px] text-muted-foreground">{t.subscriptionEnd}: {tn.subscriptionEndDate}</span>}
   310→            </div>
   311→            <div className="flex items-center justify-end gap-1 pt-2 border-t">
   312→              <ActionBtn icon={CalendarDays} label={t.extendSubscription} onClick={() => openExtend(tn)} />
   313→              <ActionBtn icon={Edit} label={t.edit} onClick={() => openEdit(tn)} />
   314→              <ActionBtn icon={tn.status === 'suspended' ? Power : PowerOff} label={tn.status === 'suspended' ? t.active : t.suspended} onClick={() => handleToggleStatus(tn)} />
   315→              <ActionBtn icon={Trash2} onClick={() => setDlg({ type: 'delete', tenant: tn })} danger />
   316→            </div>
   317→          </CardContent></Card>
   318→        ))}
   319→      </div>
   320→
   321→      <Dialog open={dlg?.type === 'add' || dlg?.type === 'edit'} onOpenChange={() => setDlg(null)}>
   322→        <DialogContent className="max-w-md"><DialogHeader><DialogTitle>{dlg?.type === 'add' ? t.addTenantDlg : t.editTenantDlg}</DialogTitle></DialogHeader>
