    77→function OverviewPage() {
    78→  const { t, isRTL, lang } = useSA()
    79→  const { sym, fmt } = useCurrency()
    80→  const [tenants] = useState<Tenant[]>(INIT_TENANTS)
    81→  const [logs] = useState<SysLog[]>(INIT_LOGS)
    82→  const [invoices] = useState<Invoice[]>(INVOICES)
    83→
    84→  // Computed from real data — no hardcoded values
    85→  const totalRevenue = tenants.reduce((s, tn) => s + tn.revenue, 0)
    86→  const totalBookings = tenants.reduce((s, tn) => s + tn.bookings, 0)
    87→  const activeCount = tenants.filter(tn => tn.status === 'active').length
    88→  const expiredCount = tenants.filter(tn => tn.subscriptionStatus === 'expired').length
    89→  const pendingSubs = tenants.filter(tn => tn.subscriptionStatus === 'trial' || tn.status === 'trial').length
    90→  const paidInvoices = invoices.filter(i => i.status === 'paid')
    91→  const monthlyRevenue = paidInvoices.reduce((s, i) => s + i.amount, 0)
    92→  const activeTenants = tenants.filter(tn => tn.status === 'active').slice(0, 6)
    93→
    94→  return (
    95→    <div className="space-y-6">
    96→      {/* Primary KPIs — all computed from tenant/invoice data */}
    97→      <div className="grid grid-cols-2 lg:grid-cols-4 gap-3 sm:gap-4">
    98→        <KpiCard icon={Building2} bg="bg-violet-600" label={t.totalTenants} value={tenants.length} sub={`${activeCount} ${t.active.toLowerCase()}`} trend={12} delay={0} />
    99→        <KpiCard icon={Users} bg="bg-sky-600" label={t.totalUsers} value={INIT_USERS.length} sub={`${tenants.reduce((s, tn) => s + tn.users, 0)} ${t.users.toLowerCase()}`} trend={8} delay={0.04} />
   100→        <KpiCard icon={Banknote} bg="bg-amber-500" label={t.revenue} value={`${(totalRevenue / 1000).toFixed(1)}K ${sym}`} sub={t.thisMonth} trend={18} delay={0.08} />
   101→        <KpiCard icon={CalendarDays} bg="bg-emerald-600" label={t.totalBookings} value={totalBookings.toLocaleString()} sub={t.last30Days} trend={5} delay={0.12} />
   102→      </div>
   103→
   104→      {/* Secondary KPIs — subscription health */}
   105→      <div className="grid grid-cols-2 lg:grid-cols-4 gap-3 sm:gap-4">
   106→        <StatCard icon={CheckCircle2} label={t.monthlyRevenue} value={fmt(monthlyRevenue)} color="text-emerald-600 bg-emerald-50 dark:bg-emerald-950/30" />
   107→        <StatCard icon={Clock} label={t.pendingRequests} value={String(pendingSubs)} color="text-amber-600 bg-amber-50 dark:bg-amber-950/30" />
   108→        <StatCard icon={AlertTriangle} label={t.expiredSubs} value={String(expiredCount)} color="text-red-600 bg-red-50 dark:bg-red-950/30" />
   109→        <StatCard icon={CreditCard} label={t.recentPayments} value={String(paidInvoices.length)} color="text-sky-600 bg-sky-50 dark:bg-sky-950/30" />
   110→      </div>
   111→
   112→      <div className="grid grid-cols-1 xl:grid-cols-3 gap-4 sm:gap-6">
   113→        <motion.div variants={fade} initial="hidden" animate="visible" transition={{ delay: 0.15 }} className="xl:col-span-2">
   114→          <Card className="border-0 shadow-sm">
   115→            <CardHeader className="pb-2"><CardTitle className="text-sm sm:text-base font-semibold">{t.revenueByPlan}</CardTitle></CardHeader>
   116→            <CardContent className="space-y-4">
   117→              {PLANS.filter(p => p.price > 0).map(p => {
   118→                const rev = tenants.filter(tn => tn.plan === p.name).reduce((s, tn) => s + tn.revenue, 0)
   119→                const pct = totalRevenue > 0 ? Math.round(rev / totalRevenue * 100) : 0
   120→                return (
   121→                  <div key={p.id} className="space-y-2">
   122→                    <div className="flex items-center justify-between text-sm">
   123→                      <span className="font-medium">{p.name}</span>
   124→                      <span className="text-muted-foreground tabular-nums text-xs sm:text-sm">{rev.toLocaleString()} {sym}</span>
   125→                    </div>
   126→                    <div className="h-2.5 rounded-full bg-muted overflow-hidden">
   127→                      <div className={`h-full rounded-full ${p.color} transition-all duration-700`} style={{ width: `${pct}%` }} />
   128→                    </div>
   129→                  </div>
   130→                )
   131→              })}
   132→            </CardContent>
   133→          </Card>
   134→        </motion.div>
   135→
   136→        <motion.div variants={fade} initial="hidden" animate="visible" transition={{ delay: 0.2 }}>
   137→          <Card className="border-0 shadow-sm h-full">
   138→            <CardHeader className="pb-2"><CardTitle className="text-sm sm:text-base font-semibold">{t.recentActivity}</CardTitle></CardHeader>
   139→            <CardContent>
   140→              <div className="space-y-1">{logs.slice(0, 6).map(l => (
   141→                <div key={l.id} className="flex gap-3 rounded-lg px-2.5 py-2 hover:bg-muted/50 transition-colors">
   142→                  <LogDot level={l.level} />
   143→                  <div className="min-w-0 flex-1">
   144→                    <p className="text-xs sm:text-sm leading-relaxed line-clamp-2">{logMsg(l, lang)}</p>
   145→                    <p className="text-[10px] sm:text-[11px] text-muted-foreground mt-0.5">{logSrc(l, lang)}</p>
   146→                  </div>
   147→                </div>
   148→              ))}</div>
   149→            </CardContent>
   150→          </Card>
   151→        </motion.div>
   152→      </div>

   160→                <button key={tn.id} className="flex items-center gap-3 rounded-xl border p-3 text-start hover:bg-accent/50 transition-colors group" onClick={() => toast.info(t.details.replace('{name}', bField(tn, 'name', lang)))}>
   161→                  <div className="flex h-10 w-10 shrink-0 items-center justify-center rounded-lg bg-violet-100 dark:bg-violet-900/30 text-violet-600 font-bold text-sm">{bField(tn, 'name', lang).charAt(0)}</div>
   162→                  <div className="min-w-0 flex-1">
   163→                    <p className="text-sm font-semibold truncate">{bField(tn, 'name', lang)}</p>

   178→// PAGE 2: TENANTS (full CRUD + sort + pagination)
   179→// ════════════════════════════════════════════════════════════════
   180→function TenantsPage() {
   181→  const { t, isRTL, lang } = useSA()
   182→  const { sym } = useCurrency()
   183→  const [tenants, setTenants] = useState<Tenant[]>(INIT_TENANTS)
   184→  const [search, setSearch] = useState('')
   185→  const [filter, setFilter] = useState('all')
   186→  const [page, setPage] = useState(1)
   187→  const [sort, setSort] = useState<SortState>({ key: 'name', dir: 'asc' })
   188→  const [dlg, setDlg] = useState<{ type: string; tenant?: Tenant } | null>(null)
   189→  const [form, setForm] = useState({ name: '', nameEn: '', email: '', country: '', plan: 'Starter' })
   190→  const [extendDlg, setExtendDlg] = useState<Tenant | null>(null)
   191→  const [extendMonths, setExtendMonths] = useState(12)
   192→  const [extendPlan, setExtendPlan] = useState('')
   193→  const [wizardOpen, setWizardOpen] = useState(false)
   194→  const [detailTenant, setDetailTenant] = useState<Tenant | null>(null)
   195→
   196→  const filtered = useMemo(() => {
   197→    let list = [...tenants]
   198→    if (filter !== 'all') list = list.filter(tn => tn.status === filter)
   199→    if (search) { const q = search.toLowerCase(); list = list.filter(tn => tn.name.includes(q) || tn.nameEn.toLowerCase().includes(q) || tn.email.toLowerCase().includes(q)) }
   200→    return genericSort(list, sort.key, sort.dir, lang)
   201→  }, [tenants, search, filter, sort, lang])
   202→
   203→  const { items, totalPages } = paginate(filtered, page, PER_PAGE)
   204→  const handleSort = (key: string) => setSort(p => p.key === key ? { key, dir: p.dir === 'asc' ? 'desc' : 'asc' } : { key, dir: 'asc' })
   205→
   206→  const openAdd = () => { setWizardOpen(true) }
   207→  const openEdit = (tn: Tenant) => { setForm({ name: tn.name, nameEn: tn.nameEn, email: tn.email, country: tn.country, plan: tn.plan }); setDlg({ type: 'edit', tenant: tn }) }
   208→
   209→  // Wizard-based tenant creation with full validation
   210→  const handleWizardSave = useCallback((data: TenantFormData) => {
   211→    // Duplicate checks
   212→    const emailExists = tenants.some(tn => tn.email.toLowerCase() === data.email.toLowerCase())
   213→    if (emailExists) { toast.error(isRTL ? 'البريد الإلكتروني مستخدم بالفعل' : 'Email already exists'); return }
   214→    const nameArExists = tenants.some(tn => tn.name === data.nameAr.trim())
   215→    if (nameArExists) { toast.error(isRTL ? 'اسم المستأجر (عربي) مستخدم بالفعل' : 'Tenant Arabic name already exists'); return }
   216→    const nameEnExists = tenants.some(tn => tn.nameEn === data.nameEn.trim())
   217→    if (nameEnExists) { toast.error(isRTL ? 'اسم المستأجر (إنجليزي) مستخدم بالفعل' : 'Tenant English name already exists'); return }
   218→
   219→    // Calculate subscription end date based on billing cycle
   220→    const cycleMonths = data.billingCycle === 'quarterly' ? 3 : data.billingCycle === 'yearly' ? 12 : 1
   221→    const endDate = new Date(data.startDate)
   222→    endDate.setMonth(endDate.getMonth() + cycleMonths)
   223→
   224→    const newTenant: Tenant = {
   225→      id: Date.now().toString(),
   226→      name: data.nameAr,
   227→      nameEn: data.nameEn,
   228→      email: data.email,
   229→      country: data.country,
   230→      plan: data.plan,
   231→      bookings: 0,
   232→      revenue: 0,
   233→      users: 1,
   234→      status: 'active',
   235→      createdAt: data.startDate,
   236→      subscriptionStatus: 'active',
   237→      subscriptionEndDate: endDate.toISOString().split('T')[0],
   238→    }
   239→    // Atomic update: add tenant + create invoice in one state batch
   240→    setTenants(prev => [newTenant, ...prev])
   241→    setPage(1)
   242→    setFilter('all')
   243→    setSearch('')
   244→  }, [tenants, isRTL])
   245→
   246→  const handleSave = useCallback(() => {
   247→    if (!form.name.trim()) { toast.error(t.enterTenantName); return }
   248→    if (form.email && !form.email.includes('@')) { toast.error(t.enterValidEmail); return }
   249→    if (dlg?.type === 'edit' && dlg.tenant) {
   250→      // Check for duplicates on edit (excluding self)
   251→      const dupEmail = tenants.some(tn => tn.id !== dlg.tenant!.id && tn.email.toLowerCase() === form.email.toLowerCase())
   252→      if (dupEmail) { toast.error(isRTL ? 'البريد الإلكتروني مستخدم بالفعل' : 'Email already exists'); return }
   253→      setTenants(p => p.map(tn => tn.id === dlg.tenant!.id ? { ...tn, ...form } : tn)); toast.success(t.updated)
   254→    }
   255→    setDlg(null); setPage(1)
   256→  }, [form, dlg, t, tenants, isRTL])
   257→
   258→  const handleDelete = (tn: Tenant) => { setTenants(p => p.filter(x => x.id !== tn.id)); toast.success(t.deleted); setPage(1) }
   259→  const handleToggleStatus = (tn: Tenant) => {
   260→    const newStatus = tn.status === 'suspended' ? 'active' : 'suspended'
   261→    setTenants(p => p.map(x => x.id === tn.id ? { ...x, status: newStatus } : x))
   262→    toast.success(newStatus === 'suspended' ? t.tenantSuspended : t.tenantActivated)
   263→    setPage(1)
   264→  }
   265→
   266→  const openExtend = (tn: Tenant) => { setExtendDlg(tn); setExtendPlan(tn.plan); setExtendMonths(12) }
   267→  const handleExtend = () => {
   268→    if (!extendDlg) return
   269→    const startDate = extendDlg.subscriptionEndDate && extendDlg.subscriptionEndDate > new Date().toISOString().split('T')[0]
   270→      ? new Date(extendDlg.subscriptionEndDate) : new Date()
   271→    const newEnd = new Date(startDate)
   272→    newEnd.setMonth(newEnd.getMonth() + extendMonths)
   273→    setTenants(p => p.map(x => x.id === extendDlg.id ? {
   274→      ...x, plan: extendPlan || x.plan,
   275→      subscriptionStatus: 'active', status: 'active',
   276→      subscriptionEndDate: newEnd.toISOString().split('T')[0]
   277→    } : x))
   278→    toast.success(t.subscriptionExtended)
   279→    setExtendDlg(null)
   280→  }
   281→
   282→  const tnName = (tn: Tenant) => bField(tn, 'name', lang)
   283→
   284→  // Existing tenants for wizard duplicate check
   285→  const existingTenants = useMemo(() => tenants.map(tn => ({ email: tn.email, name: tn.name, nameEn: tn.nameEn })), [tenants])
   286→
   287→  return (
   288→    <div className="space-y-5">
   289→      <PageTitle title={t.tenantMgmt} action={<PrimaryBtn icon={Plus} label={t.addTenant} onClick={openAdd} />} />
   290→      <div className="flex flex-col sm:flex-row gap-3">
   291→        <SearchInput value={search} onChange={v => { setSearch(v); setPage(1) }} placeholder={t.tenantSearch} />
   292→        <Select value={filter} onValueChange={v => { setFilter(v); setPage(1) }}>
   293→          <SelectTrigger className="w-full sm:w-44 h-10"><SelectValue /></SelectTrigger>
   294→          <SelectContent>
   295→            <SelectItem value="all">{t.allStatuses}</SelectItem>
   296→            <SelectItem value="active">{t.active}</SelectItem>
   297→            <SelectItem value="trial">{t.trial}</SelectItem>
   298→            <SelectItem value="suspended">{t.suspended}</SelectItem>
   299→          </SelectContent>
   300→        </Select>
   301→      </div>
   302→
   303→      <Card className="border-0 shadow-sm overflow-hidden hidden md:block">
   304→        <div className="overflow-x-auto">
   305→          <Table>
   306→            <TableHeader><TableRow className="bg-muted/50 hover:bg-muted/50">
   307→              <SortableTH label={t.tenant} sortKey="name" currentSort={sort} onSort={handleSort} className="ps-5" />
   308→              <SortableTH label={t.plan} sortKey="plan" currentSort={sort} onSort={handleSort} />
   309→              <SortableTH label={t.bookings} sortKey="bookings" currentSort={sort} onSort={handleSort} align="end" />
   310→              <SortableTH label={t.tenantRevenue} sortKey="revenue" currentSort={sort} onSort={handleSort} align="end" />
   311→              <SortableTH label={t.status} sortKey="status" currentSort={sort} onSort={handleSort} />
   312→              <TableHead className="px-3 font-semibold text-xs uppercase tracking-wider">{t.subscriptionStatus}</TableHead>
   313→              <TableHead className="pe-5 font-semibold text-xs uppercase tracking-wider text-end">{t.actions}</TableHead>
   314→            </TableRow></TableHeader>
   315→            <TableBody>
   316→              {items.length === 0 ? <EmptyRow colSpan={7} /> : items.map(tn => (
   317→                <TableRow key={tn.id} className="group hover:bg-muted/20 transition-colors">
   318→                  <TableCell className="ps-5">
   319→                    <div className="flex items-center gap-3">
   320→                      <div className="flex h-9 w-9 shrink-0 items-center justify-center rounded-lg bg-violet-100 dark:bg-violet-900/30 text-violet-600 font-bold text-xs">{tnName(tn).charAt(0)}</div>
   321→                      <div className="min-w-0"><p className="font-medium text-sm truncate max-w-[200px]">{tnName(tn)}</p><p className="text-xs text-muted-foreground truncate max-w-[200px]" dir="ltr">{tn.email}</p></div>
   322→                    </div>
   323→                  </TableCell>
   324→                  <TableCell className="px-3"><Badge variant="secondary" className="font-medium text-xs">{tn.plan}</Badge></TableCell>
   325→                  <TableCell className="px-3 text-end tabular-nums text-sm">{tn.bookings.toLocaleString()}</TableCell>
   326→                  <TableCell className="px-3 text-end font-semibold tabular-nums text-sm">{tn.revenue.toLocaleString()} <span className="text-xs font-normal text-muted-foreground">{sym}</span></TableCell>
   327→                  <TableCell className="px-3"><StatusBadge status={tn.status} locale={lang} /></TableCell>
   328→                  <TableCell className="px-3">
   329→                    <div className="space-y-0.5">
   330→                      <StatusBadge status={tn.subscriptionStatus} locale={lang} />
   331→                      {tn.subscriptionEndDate && <p className="text-[10px] text-muted-foreground">{t.subscriptionEnd}: {tn.subscriptionEndDate}</p>}
   332→                    </div>
   333→                  </TableCell>
   334→                  <TableCell className="pe-5">
   335→                    <div className="flex items-center justify-end gap-1 opacity-0 group-hover:opacity-100 transition-opacity">
   336→                      <ActionBtn icon={Eye} label={t.view} onClick={() => setDetailTenant(tn)} />
   337→                      <ActionBtn icon={Edit} label={t.edit} onClick={() => openEdit(tn)} />
   338→                      <ActionBtn icon={CalendarDays} label={t.extendSubscription} onClick={() => openExtend(tn)} />
   339→                      <ActionBtn icon={tn.status === 'suspended' ? Power : PowerOff} label={tn.status === 'suspended' ? t.active : t.suspended} onClick={() => handleToggleStatus(tn)} />
   340→                      <ActionBtn icon={Trash2} onClick={() => setDlg({ type: 'delete', tenant: tn })} danger />
   341→                    </div>
   342→                  </TableCell>
   343→                </TableRow>
   344→              ))}
   345→            </TableBody>
   346→          </Table>

   729→function BillingPage() {
   730→  const { t, lang, isRTL } = useSA()

   733→  const [filter, setFilter] = useState('all')
   734→  const [page, setPage] = useState(1)
   735→  const [sort, setSort] = useState<SortState>({ key: 'date', dir: 'desc' })
   736→  const [addOpen, setAddOpen] = useState(false)
   737→  const [invForm, setInvForm] = useState({ tenant: '', tenantEn: '', plan: 'Starter', amount: 0, status: 'pending' as string })
   738→  const [previewInv, setPreviewInv] = useState<Invoice | null>(null)
   739→
   740→  const filtered = useMemo(() => {
   741→    let list = filter === 'all' ? [...invoices] : invoices.filter(i => i.status === filter)
   742→    return genericSort(list, sort.key, sort.dir, lang)
   743→  }, [invoices, filter, sort, lang])
   744→  const { items, totalPages } = paginate(filtered, page, PER_PAGE)
   745→  const handleSort = (key: string) => setSort(p => p.key === key ? { key, dir: p.dir === 'asc' ? 'desc' : 'asc' } : { key, dir: 'asc' })
   746→  const totalPaid = invoices.filter(i => i.status === 'paid').reduce((s, i) => s + i.amount, 0)

   750→    setInvoices(p => p.map(i => {
   751→      if (i.id !== invoiceId) return i
   752→      const updated = { ...i, status: newStatus }
   753→      if (newStatus === 'paid') {
   754→        updated.paidDate = new Date().toISOString().split('T')[0]
   755→        updated.paymentMethod = method || 'cliq'
   756→        updated.history = [...(i.history || []), {
   757→          date: new Date().toISOString(), action: 'دفعة ناجحة', actionEn: 'Payment received',
   758→          details: `تم الدفع عبر ${method === 'card' ? 'بطاقة ائتمان' : method === 'bank_transfer' ? 'تحويل بنكي' : 'CLIQ'}`,

   771→  }, [])
   772→
   773→  const handleMarkPaid = (inv: Invoice) => { handleStatusChange(inv.id, 'paid') }
   774→
   775→  const handleCreateInvoice = () => {
   776→    if (!invForm.tenant.trim()) { toast.error(isRTL ? 'يرجى اختيار المستأجر' : 'Please select a tenant'); return }
   777→    if (invForm.amount <= 0) { toast.error(isRTL ? 'يرجى إدخال مبلغ صحيح' : 'Please enter a valid amount'); return }
   778→    const newId = `INV-${String(invoices.length + 1).padStart(3, '0')}`
   779→    const today = new Date().toISOString().split('T')[0]
   780→    const dueDate = new Date(); dueDate.setDate(dueDate.getDate() + 30)
   781→    const inv: Invoice = {
   782→      id: newId, tenant: invForm.tenant, tenantEn: invForm.tenantEn,

   785→      plan: invForm.plan, currency: 'JOD', taxRate: 16, notes: '', paymentMethod: '',

   787→    }
   788→    setInvoices(p => [inv, ...p]); toast.success(isRTL ? `تم إنشاء الفاتورة ${newId}` : `Invoice ${newId} created`); setAddOpen(false)
   789→    setInvForm({ tenant: '', tenantEn: '', plan: 'Starter', amount: 0, status: 'pending' })
   790→  }
   791→
   792→  const handleExportExcel = useCallback(() => {
   793→    import('xlsx').then(XLSX => {
   794→      const statusLabels: Record<string, string> = { paid: isRTL ? 'مدفوع' : 'Paid', pending: isRTL ? 'معلق' : 'Pending', overdue: isRTL ? 'متأخر' : 'Overdue', failed: isRTL ? 'فاشل' : 'Failed' }
   795→      const headers = isRTL
   796→        ? ['رقم الفاتورة', 'المستأجر', 'الباقة', 'المبلغ (د.أ)', 'العملة', 'الحالة', 'التاريخ']

   809→      toast.error(isRTL ? 'فشل تصدير Excel' : 'Excel export failed')
   810→    })
   811→  }, [filtered, isRTL])

   813→  const handleDeleteInvoice = (inv: Invoice) => {

   815→  }
   816→
   817→  return (
   818→    <div className="space-y-5">
   819→      <PageTitle title={t.billingTitle} action={
   820→        <div className="flex gap-2">
   821→          <Button variant="outline" className="gap-2 h-9" onClick={handleExportExcel}><Download className="h-4 w-4" />{t.exportExcel}</Button>

   828→        <StatCard icon={Banknote} label={t.paid} value={`${invoices.filter(i => i.status === 'paid').length}`} color="text-sky-600 bg-sky-50 dark:bg-sky-950/30" />
   829→        <StatCard icon={AlertTriangle} label={t.overdue} value={`${invoices.filter(i => i.status === 'overdue').length}`} color="text-red-600 bg-red-50 dark:bg-red-950/30" />
   830→      </div>
   831→      <div className="flex gap-3">
   832→        <Select value={filter} onValueChange={v => { setFilter(v); setPage(1) }}>
   833→          <SelectTrigger className="w-44 h-10"><SelectValue /></SelectTrigger>
   834→          <SelectContent><SelectItem value="all">{t.all}</SelectItem><SelectItem value="paid">{t.paid}</SelectItem><SelectItem value="pending">{t.pending}</SelectItem><SelectItem value="overdue">{t.overdue}</SelectItem><SelectItem value="failed">{t.failed}</SelectItem></SelectContent>
   835→        </Select>

   842→            <SortableTH label={t.tenant} sortKey="tenant" currentSort={sort} onSort={handleSort} />

   907→                </SelectContent>
   908→              </Select>
   909→            </FormField>
   910→            <div className="grid grid-cols-2 gap-3">
   911→              <FormField label={t.plan}>

   913→                  setInvForm(p => ({ ...p, plan: v }))
   914→                  const plan = PLANS.find(pl => pl.name === v)
   915→                  if (plan) setInvForm(p => ({ ...p, amount: plan.price }))
   916→                }}>
   917→                  <SelectTrigger><SelectValue /></SelectTrigger>
   918→                  <SelectContent>{PLANS.map(p => <SelectItem key={p.id} value={p.name}>{p.name} ({p.price} {sym})</SelectItem>)}</SelectContent>
   919→                </Select>
   920→              </FormField>
   921→              <FormField label={t.amount}>
   922→                <Input type="number" value={invForm.amount || ''} onChange={e => setInvForm(p => ({ ...p, amount: Number(e.target.value) }))} />
   923→              </FormField>
   924→            </div>
   925→            <FormField label={t.status}>
   926→              <Select value={invForm.status} onValueChange={v => setInvForm(p => ({ ...p, status: v }))}>
   927→                <SelectTrigger><SelectValue /></SelectTrigger>
   928→                <SelectContent>
   929→                  <SelectItem value="pending">{t.pending}</SelectItem>
   930→                  <SelectItem value="paid">{t.paid}</SelectItem>
   931→                </SelectContent>
   932→              </Select>
   933→            </FormField>
   934→          </div>
   935→          <DlgFooter onCancel={() => setAddOpen(false)} onConfirm={handleCreateInvoice} confirmLabel={isRTL ? 'إنشاء' : 'Create'} />
   936→        </DialogContent>
   937→      </Dialog>
   938→
   939→      {/* Invoice Preview Dialog */}
   940→      {previewInv && <InvoicePreviewDialog invoice={previewInv} open={!!previewInv} onClose={() => setPreviewInv(null)} onStatusChange={handleStatusChange} />}
   941→    </div>
