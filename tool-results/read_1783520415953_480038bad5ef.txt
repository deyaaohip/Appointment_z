    57→// PAGE 1: OVERVIEW (Audited — all KPIs derived from real data)

    59→function OverviewPage() {
    60→  const { t, isRTL, lang } = useSA()
    61→  const { sym, fmt } = useCurrency()
    62→  const [tenants] = useState<Tenant[]>(INIT_TENANTS)
    63→  const [logs] = useState<SysLog[]>(INIT_LOGS)
    64→  const [invoices] = useState<Invoice[]>(INVOICES)
    65→
    66→  // Computed from real data — no hardcoded values
    67→  const totalRevenue = tenants.reduce((s, tn) => s + tn.revenue, 0)
    68→  const totalBookings = tenants.reduce((s, tn) => s + tn.bookings, 0)
    69→  const activeCount = tenants.filter(tn => tn.status === 'active').length
    70→  const expiredCount = tenants.filter(tn => tn.subscriptionStatus === 'expired').length
    71→  const pendingSubs = tenants.filter(tn => tn.subscriptionStatus === 'trial' || tn.status === 'trial').length
    72→  const paidInvoices = invoices.filter(i => i.status === 'paid')
    73→  const monthlyRevenue = paidInvoices.reduce((s, i) => s + i.amount, 0)
    74→  const activeTenants = tenants.filter(tn => tn.status === 'active').slice(0, 6)
    75→
    76→  return (
    77→    <div className="space-y-6">
    78→      {/* Primary KPIs — all computed from tenant/invoice data */}
    79→      <div className="grid grid-cols-2 lg:grid-cols-4 gap-3 sm:gap-4">
    80→        <KpiCard icon={Building2} bg="bg-violet-600" label={t.totalTenants} value={tenants.length} sub={`${activeCount} ${t.active.toLowerCase()}`} trend={12} delay={0} />
    81→        <KpiCard icon={Users} bg="bg-sky-600" label={t.totalUsers} value={INIT_USERS.length} sub={`${tenants.reduce((s, tn) => s + tn.users, 0)} ${t.users.toLowerCase()}`} trend={8} delay={0.04} />
    82→        <KpiCard icon={DollarSign} bg="bg-amber-500" label={t.revenue} value={`${(totalRevenue / 1000).toFixed(1)}K ${sym}`} sub={t.thisMonth} trend={18} delay={0.08} />
    83→        <KpiCard icon={CalendarDays} bg="bg-emerald-600" label={t.totalBookings} value={totalBookings.toLocaleString()} sub={t.last30Days} trend={5} delay={0.12} />
    84→      </div>
    85→
    86→      {/* Secondary KPIs — subscription health */}
    87→      <div className="grid grid-cols-2 lg:grid-cols-4 gap-3 sm:gap-4">
    88→        <StatCard icon={CheckCircle2} label={t.monthlyRevenue} value={fmt(monthlyRevenue)} color="text-emerald-600 bg-emerald-50 dark:bg-emerald-950/30" />
    89→        <StatCard icon={Clock} label={t.pendingRequests} value={String(pendingSubs)} color="text-amber-600 bg-amber-50 dark:bg-amber-950/30" />
    90→        <StatCard icon={AlertTriangle} label={t.expiredSubs} value={String(expiredCount)} color="text-red-600 bg-red-50 dark:bg-red-950/30" />
    91→        <StatCard icon={CreditCard} label={t.recentPayments} value={String(paidInvoices.length)} color="text-sky-600 bg-sky-50 dark:bg-sky-950/30" />
    92→      </div>
    93→
    94→      <div className="grid grid-cols-1 xl:grid-cols-3 gap-4 sm:gap-6">
    95→        <motion.div variants={fade} initial="hidden" animate="visible" transition={{ delay: 0.15 }} className="xl:col-span-2">
    96→          <Card className="border-0 shadow-sm">
    97→            <CardHeader className="pb-2"><CardTitle className="text-sm sm:text-base font-semibold">{t.revenueByPlan}</CardTitle></CardHeader>
    98→            <CardContent className="space-y-4">
    99→              {PLANS.filter(p => p.price > 0).map(p => {
   100→                const rev = tenants.filter(tn => tn.plan === p.name).reduce((s, tn) => s + tn.revenue, 0)
   101→                const pct = totalRevenue > 0 ? Math.round(rev / totalRevenue * 100) : 0
   102→                return (
   103→                  <div key={p.id} className="space-y-2">
   104→                    <div className="flex items-center justify-between text-sm">
   105→                      <span className="font-medium">{p.name}</span>
   106→                      <span className="text-muted-foreground tabular-nums text-xs sm:text-sm">{rev.toLocaleString()} {sym}</span>
   107→                    </div>
   108→                    <div className="h-2.5 rounded-full bg-muted overflow-hidden">
   109→                      <div className={`h-full rounded-full ${p.color} transition-all duration-700`} style={{ width: `${pct}%` }} />
   110→                    </div>
   111→                  </div>
   112→                )
   113→              })}
   114→            </CardContent>
   115→          </Card>
   116→        </motion.div>
   117→
   118→        <motion.div variants={fade} initial="hidden" animate="visible" transition={{ delay: 0.2 }}>
   119→          <Card className="border-0 shadow-sm h-full">
   120→            <CardHeader className="pb-2"><CardTitle className="text-sm sm:text-base font-semibold">{t.recentActivity}</CardTitle></CardHeader>
   121→            <CardContent>
   122→              <div className="space-y-1">{logs.slice(0, 6).map(l => (
   123→                <div key={l.id} className="flex gap-3 rounded-lg px-2.5 py-2 hover:bg-muted/50 transition-colors">

   160→// PAGE 2: TENANTS (full CRUD + sort + pagination)
   161→// ════════════════════════════════════════════════════════════════
   162→function TenantsPage() {
   163→  const { t, isRTL, lang } = useSA()
   164→  const { sym } = useCurrency()
   165→  const [tenants, setTenants] = useState<Tenant[]>(INIT_TENANTS)
   166→  const [search, setSearch] = useState('')
   167→  const [filter, setFilter] = useState('all')
   168→  const [page, setPage] = useState(1)
   169→  const [sort, setSort] = useState<SortState>({ key: 'name', dir: 'asc' })
   170→  const [dlg, setDlg] = useState<{ type: string; tenant?: Tenant } | null>(null)
   171→  const [form, setForm] = useState({ name: '', nameEn: '', email: '', country: '', plan: 'Starter' })
   172→
   173→  const filtered = useMemo(() => {
   174→    let list = [...tenants]
   175→    if (filter !== 'all') list = list.filter(tn => tn.status === filter)
   176→    if (search) { const q = search.toLowerCase(); list = list.filter(tn => tn.name.includes(q) || tn.nameEn.toLowerCase().includes(q) || tn.email.toLowerCase().includes(q)) }
   177→    return genericSort(list, sort.key, sort.dir, lang)
   178→  }, [tenants, search, filter, sort, lang])
   179→
   180→  const { items, totalPages } = paginate(filtered, page, PER_PAGE)
   181→  const handleSort = (key: string) => setSort(p => p.key === key ? { key, dir: p.dir === 'asc' ? 'desc' : 'asc' } : { key, dir: 'asc' })
   182→
   183→  const openAdd = () => { setForm({ name: '', nameEn: '', email: '', country: '', plan: 'Starter' }); setDlg({ type: 'add' }) }
   184→  const openEdit = (tn: Tenant) => { setForm({ name: tn.name, nameEn: tn.nameEn, email: tn.email, country: tn.country, plan: tn.plan }); setDlg({ type: 'edit', tenant: tn }) }
   185→
   186→  const handleSave = useCallback(() => {
   187→    if (!form.name.trim()) { toast.error(t.enterTenantName); return }
   188→    if (form.email && !form.email.includes('@')) { toast.error(t.enterValidEmail); return }
   189→    if (dlg?.type === 'add') {
   190→      const n: Tenant = { id: Date.now().toString(), ...form, bookings: 0, revenue: 0, users: 1, status: 'trial', createdAt: new Date().toISOString().split('T')[0], subscriptionStatus: 'trial', subscriptionEndDate: '' }
   191→      setTenants(p => [n, ...p]); toast.success(t.added.replace('{name}', form.name))
   192→    } else if (dlg?.tenant) {
   193→      setTenants(p => p.map(tn => tn.id === dlg.tenant!.id ? { ...tn, ...form } : tn)); toast.success(t.updated)
   194→    }
   195→    setDlg(null); setPage(1)
   196→  }, [form, dlg, t])
   197→
   198→  const handleDelete = (tn: Tenant) => { setTenants(p => p.filter(x => x.id !== tn.id)); toast.success(t.deleted); setPage(1) }
   199→  const handleToggleStatus = (tn: Tenant) => {
   200→    const newStatus = tn.status === 'suspended' ? 'active' : 'suspended'
   201→    setTenants(p => p.map(x => x.id === tn.id ? { ...x, status: newStatus } : x))
   202→    toast.success(newStatus === 'suspended' ? t.tenantSuspended : t.tenantActivated)
   203→    setPage(1)
   204→  }
   205→
   206→  const tnName = (tn: Tenant) => bField(tn, 'name', lang)
   207→
   208→  return (
   209→    <div className="space-y-5">
   210→      <PageTitle title={t.tenantMgmt} action={<PrimaryBtn icon={Plus} label={t.addTenant} onClick={openAdd} />} />
   211→      <div className="flex flex-col sm:flex-row gap-3">
   212→        <SearchInput value={search} onChange={v => { setSearch(v); setPage(1) }} placeholder={t.tenantSearch} />
   213→        <Select value={filter} onValueChange={v => { setFilter(v); setPage(1) }}>
   214→          <SelectTrigger className="w-full sm:w-44 h-10"><SelectValue /></SelectTrigger>
   215→          <SelectContent>
   216→            <SelectItem value="all">{t.allStatuses}</SelectItem>
   217→            <SelectItem value="active">{t.active}</SelectItem>
   218→            <SelectItem value="trial">{t.trial}</SelectItem>
   219→            <SelectItem value="suspended">{t.suspended}</SelectItem>
   220→          </SelectContent>
   221→        </Select>
   222→      </div>
   223→
   224→      <Card className="border-0 shadow-sm overflow-hidden hidden md:block">
   225→        <div className="overflow-x-auto">
   226→          <Table>
   227→            <TableHeader><TableRow className="bg-muted/50 hover:bg-muted/50">
   228→              <SortableTH label={t.tenant} sortKey="name" currentSort={sort} onSort={handleSort} className="ps-5" />
   229→              <SortableTH label={t.plan} sortKey="plan" currentSort={sort} onSort={handleSort} />
   230→              <SortableTH label={t.bookings} sortKey="bookings" currentSort={sort} onSort={handleSort} align="end" />
   231→              <SortableTH label={t.tenantRevenue} sortKey="revenue" currentSort={sort} onSort={handleSort} align="end" />
   232→              <SortableTH label={t.status} sortKey="status" currentSort={sort} onSort={handleSort} />
   233→              <TableHead className="pe-5 font-semibold text-xs uppercase tracking-wider text-end">{t.actions}</TableHead>
   234→            </TableRow></TableHeader>
   235→            <TableBody>
   236→              {items.length === 0 ? <EmptyRow colSpan={6} /> : items.map(tn => (
   237→                <TableRow key={tn.id} className="group hover:bg-muted/20 transition-colors">
   238→                  <TableCell className="ps-5">
   239→                    <div className="flex items-center gap-3">
   240→                      <div className="flex h-9 w-9 shrink-0 items-center justify-center rounded-lg bg-violet-100 dark:bg-violet-900/30 text-violet-600 font-bold text-xs">{tnName(tn).charAt(0)}</div>
   241→                      <div className="min-w-0"><p className="font-medium text-sm truncate max-w-[200px]">{tnName(tn)}</p><p className="text-xs text-muted-foreground truncate max-w-[200px]" dir="ltr">{tn.email}</p></div>
   242→                    </div>
   243→                  </TableCell>
   244→                  <TableCell className="px-3"><Badge variant="secondary" className="font-medium text-xs">{tn.plan}</Badge></TableCell>
   245→                  <TableCell className="px-3 text-end tabular-nums text-sm">{tn.bookings.toLocaleString()}</TableCell>
   246→                  <TableCell className="px-3 text-end font-semibold tabular-nums text-sm">{tn.revenue.toLocaleString()} <span className="text-xs font-normal text-muted-foreground">{sym}</span></TableCell>
   247→                  <TableCell className="px-3"><StatusBadge status={tn.status} locale={lang} /></TableCell>
   248→                  <TableCell className="pe-5">
   249→                    <div className="flex items-center justify-end gap-1 opacity-0 group-hover:opacity-100 transition-opacity">
   250→                      <ActionBtn icon={Eye} label={t.view} onClick={() => toast.info(t.details.replace('{name}', tnName(tn)))} />
   251→                      <ActionBtn icon={Edit} label={t.edit} onClick={() => openEdit(tn)} />
   252→                      <ActionBtn icon={tn.status === 'suspended' ? Power : PowerOff} label={tn.status === 'suspended' ? t.active : t.suspended} onClick={() => handleToggleStatus(tn)} />
   253→                      <ActionBtn icon={Trash2} onClick={() => setDlg({ type: 'delete', tenant: tn })} danger />
   254→                    </div>
   255→                  </TableCell>
   256→                </TableRow>
   257→              ))}
   258→            </TableBody>
   259→          </Table>
   260→        </div>
   261→        <TableFooter current={items.length} total={filtered.length} entity={t.tenant.toLowerCase()} />
   262→        <Pagination page={page} totalPages={totalPages} onPageChange={setPage} />
   263→      </Card>
   264→
   265→      <div className="md:hidden space-y-3">
   266→        {items.length === 0 ? <Card className="border-0 shadow-sm"><CardContent className="py-16 text-center text-muted-foreground text-sm">{t.noResults}</CardContent></Card> : items.map(tn => (
   267→          <Card key={tn.id} className="border-0 shadow-sm"><CardContent className="p-4">
   268→            <div className="flex items-start justify-between gap-3 mb-3">
   269→              <div className="flex items-center gap-3 min-w-0">
   270→                <div className="flex h-10 w-10 shrink-0 items-center justify-center rounded-lg bg-violet-100 dark:bg-violet-900/30 text-violet-600 font-bold text-sm">{tnName(tn).charAt(0)}</div>
   271→                <div className="min-w-0"><p className="font-semibold text-sm truncate">{tnName(tn)}</p><p className="text-xs text-muted-foreground truncate" dir="ltr">{tn.email}</p></div>
   272→              </div>
   273→              <StatusBadge status={tn.status} locale={lang} />
   274→            </div>
   275→            <div className="grid grid-cols-3 gap-3 mb-3 text-center">
   276→              <div className="rounded-lg bg-muted/50 p-2"><p className="text-[10px] text-muted-foreground">{t.plan}</p><p className="text-xs font-semibold mt-0.5">{tn.plan}</p></div>
   277→              <div className="rounded-lg bg-muted/50 p-2"><p className="text-[10px] text-muted-foreground">{t.bookings}</p><p className="text-xs font-semibold mt-0.5 tabular-nums">{tn.bookings.toLocaleString()}</p></div>

   280→            <div className="flex items-center justify-end gap-1 pt-2 border-t">
   281→              <ActionBtn icon={Edit} label={t.edit} onClick={() => openEdit(tn)} />

   283→              <ActionBtn icon={Trash2} onClick={() => setDlg({ type: 'delete', tenant: tn })} danger />
   284→            </div>
   285→          </CardContent></Card>
   286→        ))}
   287→      </div>
   288→
   289→      <Dialog open={dlg?.type === 'add' || dlg?.type === 'edit'} onOpenChange={() => setDlg(null)}>
   290→        <DialogContent className="max-w-md"><DialogHeader><DialogTitle>{dlg?.type === 'add' ? t.addTenantDlg : t.editTenantDlg}</DialogTitle></DialogHeader>
   291→          <div className="space-y-4 pt-2">
   292→            <FormField label={t.nameAr}><Input value={form.name} onChange={e => setForm(p => ({ ...p, name: e.target.value }))} /></FormField>
   293→            <FormField label={t.nameEn}><Input value={form.nameEn} onChange={e => setForm(p => ({ ...p, nameEn: e.target.value }))} dir="ltr" /></FormField>

   295→            <div className="grid grid-cols-2 gap-3">
   296→              <FormField label={t.country}><Input value={form.country} onChange={e => setForm(p => ({ ...p, country: e.target.value }))} /></FormField>

   299→          </div>
   300→          <DlgFooter onCancel={() => setDlg(null)} onConfirm={handleSave} />

   302→      </Dialog>
   303→      <ConfirmDialog open={dlg?.type === 'delete'} onOpenChange={() => setDlg(null)} title={t.deleteTenantDlg} desc={t.deleteTenantDesc.replace('{name}', dlg?.tenant?.name || '')} onConfirm={() => { if (dlg?.tenant) handleDelete(dlg.tenant) }} danger />
   304→    </div>
   305→  )
