    64→function OverviewPage() {
    65→  const { t, isRTL, lang } = useSA()

    69→  const [invoices] = useState<Invoice[]>(INVOICES)
    70→
    71→  // Computed from real data — no hardcoded values
    72→  const totalRevenue = tenants.reduce((s, tn) => s + tn.revenue, 0)
    73→  const totalBookings = tenants.reduce((s, tn) => s + tn.bookings, 0)
    74→  const activeCount = tenants.filter(tn => tn.status === 'active').length
    75→  const expiredCount = tenants.filter(tn => tn.subscriptionStatus === 'expired').length
    76→  const pendingSubs = tenants.filter(tn => tn.subscriptionStatus === 'trial' || tn.status === 'trial').length
    77→  const paidInvoices = invoices.filter(i => i.status === 'paid')
    78→  const monthlyRevenue = paidInvoices.reduce((s, i) => s + i.amount, 0)
    79→  const activeTenants = tenants.filter(tn => tn.status === 'active').slice(0, 6)
    80→
    81→  return (
    82→    <div className="space-y-6">
    83→      {/* Primary KPIs — all computed from tenant/invoice data */}
    84→      <div className="grid grid-cols-2 lg:grid-cols-4 gap-3 sm:gap-4">
    85→        <KpiCard icon={Building2} bg="bg-violet-600" label={t.totalTenants} value={tenants.length} sub={`${activeCount} ${t.active.toLowerCase()}`} trend={12} delay={0} />
    86→        <KpiCard icon={Users} bg="bg-sky-600" label={t.totalUsers} value={INIT_USERS.length} sub={`${tenants.reduce((s, tn) => s + tn.users, 0)} ${t.users.toLowerCase()}`} trend={8} delay={0.04} />
    87→        <KpiCard icon={Banknote} bg="bg-amber-500" label={t.revenue} value={`${(totalRevenue / 1000).toFixed(1)}K ${sym}`} sub={t.thisMonth} trend={18} delay={0.08} />
    88→        <KpiCard icon={CalendarDays} bg="bg-emerald-600" label={t.totalBookings} value={totalBookings.toLocaleString()} sub={t.last30Days} trend={5} delay={0.12} />
    89→      </div>
    90→
    91→      {/* Secondary KPIs — subscription health */}
    92→      <div className="grid grid-cols-2 lg:grid-cols-4 gap-3 sm:gap-4">
    93→        <StatCard icon={CheckCircle2} label={t.monthlyRevenue} value={fmt(monthlyRevenue)} color="text-emerald-600 bg-emerald-50 dark:bg-emerald-950/30" />
    94→        <StatCard icon={Clock} label={t.pendingRequests} value={String(pendingSubs)} color="text-amber-600 bg-amber-50 dark:bg-amber-950/30" />
    95→        <StatCard icon={AlertTriangle} label={t.expiredSubs} value={String(expiredCount)} color="text-red-600 bg-red-50 dark:bg-red-950/30" />
    96→        <StatCard icon={CreditCard} label={t.recentPayments} value={String(paidInvoices.length)} color="text-sky-600 bg-sky-50 dark:bg-sky-950/30" />
    97→      </div>
    98→
    99→      <div className="grid grid-cols-1 xl:grid-cols-3 gap-4 sm:gap-6">
   100→        <motion.div variants={fade} initial="hidden" animate="visible" transition={{ delay: 0.15 }} className="xl:col-span-2">
   101→          <Card className="border-0 shadow-sm">
   102→            <CardHeader className="pb-2"><CardTitle className="text-sm sm:text-base font-semibold">{t.revenueByPlan}</CardTitle></CardHeader>
   103→            <CardContent className="space-y-4">
   104→              {PLANS.filter(p => p.price > 0).map(p => {
   105→                const rev = tenants.filter(tn => tn.plan === p.name).reduce((s, tn) => s + tn.revenue, 0)
   106→                const pct = totalRevenue > 0 ? Math.round(rev / totalRevenue * 100) : 0
   107→                return (
   108→                  <div key={p.id} className="space-y-2">
   109→                    <div className="flex items-center justify-between text-sm">
   110→                      <span className="font-medium">{p.name}</span>
   111→                      <span className="text-muted-foreground tabular-nums text-xs sm:text-sm">{rev.toLocaleString()} {sym}</span>
   112→                    </div>
   113→                    <div className="h-2.5 rounded-full bg-muted overflow-hidden">
   114→                      <div className={`h-full rounded-full ${p.color} transition-all duration-700`} style={{ width: `${pct}%` }} />
   115→                    </div>
   116→                  </div>
   117→                )
   118→              })}
   119→            </CardContent>
   120→          </Card>
   121→        </motion.div>
   122→
   123→        <motion.div variants={fade} initial="hidden" animate="visible" transition={{ delay: 0.2 }}>
   124→          <Card className="border-0 shadow-sm h-full">
   125→            <CardHeader className="pb-2"><CardTitle className="text-sm sm:text-base font-semibold">{t.recentActivity}</CardTitle></CardHeader>
   126→            <CardContent>
   127→              <div className="space-y-1">{logs.slice(0, 6).map(l => (
   128→                <div key={l.id} className="flex gap-3 rounded-lg px-2.5 py-2 hover:bg-muted/50 transition-colors">
   129→                  <LogDot level={l.level} />

   165→// PAGE 2: TENANTS (full CRUD + sort + pagination)

   167→function TenantsPage() {
   168→  const { t, isRTL, lang } = useSA()
   169→  const { sym } = useCurrency()
   170→  const [tenants, setTenants] = useState<Tenant[]>(INIT_TENANTS)
   171→  const [search, setSearch] = useState('')
   172→  const [filter, setFilter] = useState('all')
   173→  const [page, setPage] = useState(1)
   174→  const [sort, setSort] = useState<SortState>({ key: 'name', dir: 'asc' })
   175→  const [dlg, setDlg] = useState<{ type: string; tenant?: Tenant } | null>(null)
   176→  const [form, setForm] = useState({ name: '', nameEn: '', email: '', country: '', plan: 'Starter' })
   177→  const [extendDlg, setExtendDlg] = useState<Tenant | null>(null)
   178→  const [extendMonths, setExtendMonths] = useState(12)
   179→  const [extendPlan, setExtendPlan] = useState('')
   180→  const [wizardOpen, setWizardOpen] = useState(false)
   181→  const [detailTenant, setDetailTenant] = useState<Tenant | null>(null)
   182→
   183→  const filtered = useMemo(() => {
   184→    let list = [...tenants]
   185→    if (filter !== 'all') list = list.filter(tn => tn.status === filter)
   186→    if (search) { const q = search.toLowerCase(); list = list.filter(tn => tn.name.includes(q) || tn.nameEn.toLowerCase().includes(q) || tn.email.toLowerCase().includes(q)) }
   187→    return genericSort(list, sort.key, sort.dir, lang)
   188→  }, [tenants, search, filter, sort, lang])
   189→
   190→  const { items, totalPages } = paginate(filtered, page, PER_PAGE)
   191→  const handleSort = (key: string) => setSort(p => p.key === key ? { key, dir: p.dir === 'asc' ? 'desc' : 'asc' } : { key, dir: 'asc' })
   192→
   193→  const openAdd = () => { setWizardOpen(true) }
   194→  const openEdit = (tn: Tenant) => { setForm({ name: tn.name, nameEn: tn.nameEn, email: tn.email, country: tn.country, plan: tn.plan }); setDlg({ type: 'edit', tenant: tn }) }
   195→
   196→  // Wizard-based tenant creation with full validation
   197→  const handleWizardSave = useCallback((data: TenantFormData) => {
   198→    // Duplicate checks
   199→    const emailExists = tenants.some(tn => tn.email.toLowerCase() === data.email.toLowerCase())
   200→    if (emailExists) { toast.error(isRTL ? 'البريد الإلكتروني مستخدم بالفعل' : 'Email already exists'); return }
   201→    const nameArExists = tenants.some(tn => tn.name === data.nameAr.trim())
   202→    if (nameArExists) { toast.error(isRTL ? 'اسم المستأجر (عربي) مستخدم بالفعل' : 'Tenant Arabic name already exists'); return }
   203→    const nameEnExists = tenants.some(tn => tn.nameEn === data.nameEn.trim())
   204→    if (nameEnExists) { toast.error(isRTL ? 'اسم المستأجر (إنجليزي) مستخدم بالفعل' : 'Tenant English name already exists'); return }
   205→
   206→    // Calculate subscription end date based on billing cycle
   207→    const cycleMonths = data.billingCycle === 'quarterly' ? 3 : data.billingCycle === 'yearly' ? 12 : 1
   208→    const endDate = new Date(data.startDate)
   209→    endDate.setMonth(endDate.getMonth() + cycleMonths)
   210→
   211→    const newTenant: Tenant = {
   212→      id: Date.now().toString(),
   213→      name: data.nameAr,
   214→      nameEn: data.nameEn,
   215→      email: data.email,
   216→      country: data.country,
   217→      plan: data.plan,
   218→      bookings: 0,
   219→      revenue: 0,
   220→      users: 1,
   221→      status: 'active',
   222→      createdAt: data.startDate,
   223→      subscriptionStatus: 'active',
   224→      subscriptionEndDate: endDate.toISOString().split('T')[0],
   225→    }
   226→    // Atomic update: add tenant + create invoice in one state batch
   227→    setTenants(prev => [newTenant, ...prev])
   228→    setPage(1)
   229→    setFilter('all')
   230→    setSearch('')
   231→  }, [tenants, isRTL])
   232→
   233→  const handleSave = useCallback(() => {
   234→    if (!form.name.trim()) { toast.error(t.enterTenantName); return }
   235→    if (form.email && !form.email.includes('@')) { toast.error(t.enterValidEmail); return }
   236→    if (dlg?.type === 'edit' && dlg.tenant) {
   237→      // Check for duplicates on edit (excluding self)
   238→      const dupEmail = tenants.some(tn => tn.id !== dlg.tenant!.id && tn.email.toLowerCase() === form.email.toLowerCase())
   239→      if (dupEmail) { toast.error(isRTL ? 'البريد الإلكتروني مستخدم بالفعل' : 'Email already exists'); return }
   240→      setTenants(p => p.map(tn => tn.id === dlg.tenant!.id ? { ...tn, ...form } : tn)); toast.success(t.updated)
   241→    }
   242→    setDlg(null); setPage(1)
   243→  }, [form, dlg, t, tenants, isRTL])
   244→
   245→  const handleDelete = (tn: Tenant) => { setTenants(p => p.filter(x => x.id !== tn.id)); toast.success(t.deleted); setPage(1) }
   246→  const handleToggleStatus = (tn: Tenant) => {
   247→    const newStatus = tn.status === 'suspended' ? 'active' : 'suspended'
   248→    setTenants(p => p.map(x => x.id === tn.id ? { ...x, status: newStatus } : x))
   249→    toast.success(newStatus === 'suspended' ? t.tenantSuspended : t.tenantActivated)
   250→    setPage(1)
   251→  }
   252→
   253→  const openExtend = (tn: Tenant) => { setExtendDlg(tn); setExtendPlan(tn.plan); setExtendMonths(12) }
   254→  const handleExtend = () => {
   255→    if (!extendDlg) return
   256→    const startDate = extendDlg.subscriptionEndDate && extendDlg.subscriptionEndDate > new Date().toISOString().split('T')[0]
   257→      ? new Date(extendDlg.subscriptionEndDate) : new Date()
   258→    const newEnd = new Date(startDate)
   259→    newEnd.setMonth(newEnd.getMonth() + extendMonths)
   260→    setTenants(p => p.map(x => x.id === extendDlg.id ? {
   261→      ...x, plan: extendPlan || x.plan,
   262→      subscriptionStatus: 'active', status: 'active',
   263→      subscriptionEndDate: newEnd.toISOString().split('T')[0]
   264→    } : x))
   265→    toast.success(t.subscriptionExtended)
   266→    setExtendDlg(null)
   267→  }
   268→
   269→  const tnName = (tn: Tenant) => bField(tn, 'name', lang)
   270→
   271→  // Existing tenants for wizard duplicate check
   272→  const existingTenants = useMemo(() => tenants.map(tn => ({ email: tn.email, name: tn.name, nameEn: tn.nameEn })), [tenants])
   273→
   274→  return (
   275→    <div className="space-y-5">
   276→      <PageTitle title={t.tenantMgmt} action={<PrimaryBtn icon={Plus} label={t.addTenant} onClick={openAdd} />} />
   277→      <div className="flex flex-col sm:flex-row gap-3">
   278→        <SearchInput value={search} onChange={v => { setSearch(v); setPage(1) }} placeholder={t.tenantSearch} />
   279→        <Select value={filter} onValueChange={v => { setFilter(v); setPage(1) }}>
   280→          <SelectTrigger className="w-full sm:w-44 h-10"><SelectValue /></SelectTrigger>
   281→          <SelectContent>
   282→            <SelectItem value="all">{t.allStatuses}</SelectItem>
   283→            <SelectItem value="active">{t.active}</SelectItem>
   284→            <SelectItem value="trial">{t.trial}</SelectItem>
   285→            <SelectItem value="suspended">{t.suspended}</SelectItem>
   286→          </SelectContent>
   287→        </Select>
   288→      </div>
   289→
   290→      <Card className="border-0 shadow-sm overflow-hidden hidden md:block">
   291→        <div className="overflow-x-auto">
   292→          <Table>
   293→            <TableHeader><TableRow className="bg-muted/50 hover:bg-muted/50">
   294→              <SortableTH label={t.tenant} sortKey="name" currentSort={sort} onSort={handleSort} className="ps-5" />
   295→              <SortableTH label={t.plan} sortKey="plan" currentSort={sort} onSort={handleSort} />
   296→              <SortableTH label={t.bookings} sortKey="bookings" currentSort={sort} onSort={handleSort} align="end" />
   297→              <SortableTH label={t.tenantRevenue} sortKey="revenue" currentSort={sort} onSort={handleSort} align="end" />
   298→              <SortableTH label={t.status} sortKey="status" currentSort={sort} onSort={handleSort} />
   299→              <TableHead className="px-3 font-semibold text-xs uppercase tracking-wider">{t.subscriptionStatus}</TableHead>
   300→              <TableHead className="pe-5 font-semibold text-xs uppercase tracking-wider text-end">{t.actions}</TableHead>
   301→            </TableRow></TableHeader>
   302→            <TableBody>
   303→              {items.length === 0 ? <EmptyRow colSpan={7} /> : items.map(tn => (
   304→                <TableRow key={tn.id} className="group hover:bg-muted/20 transition-colors">
   305→                  <TableCell className="ps-5">
   306→                    <div className="flex items-center gap-3">
   307→                      <div className="flex h-9 w-9 shrink-0 items-center justify-center rounded-lg bg-violet-100 dark:bg-violet-900/30 text-violet-600 font-bold text-xs">{tnName(tn).charAt(0)}</div>
   308→                      <div className="min-w-0"><p className="font-medium text-sm truncate max-w-[200px]">{tnName(tn)}</p><p className="text-xs text-muted-foreground truncate max-w-[200px]" dir="ltr">{tn.email}</p></div>
   309→                    </div>
   310→                  </TableCell>
   311→                  <TableCell className="px-3"><Badge variant="secondary" className="font-medium text-xs">{tn.plan}</Badge></TableCell>
   312→                  <TableCell className="px-3 text-end tabular-nums text-sm">{tn.bookings.toLocaleString()}</TableCell>
   313→                  <TableCell className="px-3 text-end font-semibold tabular-nums text-sm">{tn.revenue.toLocaleString()} <span className="text-xs font-normal text-muted-foreground">{sym}</span></TableCell>
   314→                  <TableCell className="px-3"><StatusBadge status={tn.status} locale={lang} /></TableCell>
   315→                  <TableCell className="px-3">
   316→                    <div className="space-y-0.5">
   317→                      <StatusBadge status={tn.subscriptionStatus} locale={lang} />
   318→                      {tn.subscriptionEndDate && <p className="text-[10px] text-muted-foreground">{t.subscriptionEnd}: {tn.subscriptionEndDate}</p>}
   319→                    </div>
   320→                  </TableCell>
   321→                  <TableCell className="pe-5">
   322→                    <div className="flex items-center justify-end gap-1 opacity-0 group-hover:opacity-100 transition-opacity">
   323→                      <ActionBtn icon={Eye} label={t.view} onClick={() => setDetailTenant(tn)} />
   324→                      <ActionBtn icon={Edit} label={t.edit} onClick={() => openEdit(tn)} />
   325→                      <ActionBtn icon={CalendarDays} label={t.extendSubscription} onClick={() => openExtend(tn)} />
   326→                      <ActionBtn icon={tn.status === 'suspended' ? Power : PowerOff} label={tn.status === 'suspended' ? t.active : t.suspended} onClick={() => handleToggleStatus(tn)} />
   327→                      <ActionBtn icon={Trash2} onClick={() => setDlg({ type: 'delete', tenant: tn })} danger />
   328→                    </div>
   329→                  </TableCell>
   330→                </TableRow>
   331→              ))}
   332→            </TableBody>
   333→          </Table>
   334→        </div>
   335→        <TableFooter current={items.length} total={filtered.length} entity={t.tenant.toLowerCase()} />
   336→        <Pagination page={page} totalPages={totalPages} onPageChange={setPage} />
   337→      </Card>
   338→
   339→      <div className="md:hidden space-y-3">
   340→        {items.length === 0 ? <Card className="border-0 shadow-sm"><CardContent className="py-16 text-center text-muted-foreground text-sm">{t.noResults}</CardContent></Card> : items.map(tn => (
   341→          <Card key={tn.id} className="border-0 shadow-sm"><CardContent className="p-4">
