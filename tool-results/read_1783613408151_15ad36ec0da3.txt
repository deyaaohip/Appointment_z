    17→    17→// ─── Simple QR Code SVG Generator ──────────────────────────────
    18→    18→function SimpleQRCode({ data, size = 120 }: { data: string; size?: number }) {
    19→    19→  const hash = Array.from(data).reduce((acc, c) => ((acc << 5) - acc + c.charCodeAt(0)) | 0, 0)
    20→    20→  const cells = 21
    21→    21→  const matrix: boolean[][] = Array.from({ length: cells }, () => Array(cells).fill(false))
    22→    22→  const seed = Math.abs(hash)
    23→    23→  const rng = (i: number) => { let x = Math.sin(seed + i * 127.1) * 43758.5453; return x - Math.floor(x) }
    24→    24→  const drawFinder = (r: number, c: number) => {
    25→    25→    for (let i = 0; i < 7; i++) for (let j = 0; j < 7; j++) {
    26→    26→      const isBorder = i === 0 || i === 6 || j === 0 || j === 6
    27→    27→      const isInner = i >= 2 && i <= 4 && j >= 2 && j <= 4
    28→    28→      matrix[r + i][c + j] = isBorder || isInner
    29→    29→    }
    30→    30→  }
    31→    31→  drawFinder(0, 0); drawFinder(0, 14); drawFinder(14, 0)
    32→    32→  let idx = 0
    33→    33→  for (let r = 0; r < cells; r++) for (let c = 0; c < cells; c++) {
    34→    34→    if (matrix[r][c]) continue
    35→    35→    const inTL = r < 8 && c < 8; const inTR = r < 8 && c > 12; const inBL = r > 12 && c < 8
    36→    36→    if (inTL || inTR || inBL) continue
    37→    37→    if (rng(idx++) > 0.5) matrix[r][c] = true
    38→    38→  }
    39→    39→  const cellSize = size / cells
    40→    40→  return (
    41→    41→    <svg width={size} height={size} viewBox={`0 0 ${size} ${size}`} className="mx-auto">
    42→    42→      <rect width={size} height={size} fill="white" rx="4" />
    43→    43→      {matrix.map((row, r) => row.map((filled, c) => filled ? (
    44→    44→        <rect key={`${r}-${c}`} x={c * cellSize} y={r * cellSize} width={cellSize} height={cellSize} fill="#1a1a2e" rx={0.5} />
    45→    45→      ) : null))}
    46→    46→    </svg>
    47→    47→  )
    48→    48→}

    51→    51→function payMethodLabel(method: string, lang: 'ar' | 'en'): string {
    52→    52→  if (!method) return lang === 'ar' ? '—' : '—'
    53→    53→  const map: Record<string, { ar: string; en: string }> = {
    54→    54→    cliq: { ar: 'CLIQ تحويل', en: 'CLIQ Transfer' },
    55→    55→    card: { ar: 'بطاقة ائتمان', en: 'Credit Card' },
    56→    56→    bank_transfer: { ar: 'تحويل بنكي', en: 'Bank Transfer' },
    57→    57→  }
    58→    58→  return map[method]?.[lang] || method
    59→    59→}
